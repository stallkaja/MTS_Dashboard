CREATE TABLE `caltoolstable` (
   `ID` varchar(255) NOT NULL,
   `ManufacturerName` text,
   `ModelName` text,
   `Description` text,
   `SerialNumber` int DEFAULT NULL,
   `Area` int NOT NULL,
   `Location` text,
   `CalibrationDue` text,
   PRIMARY KEY (`ID`,`Area`)
 ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci

   `MaterialName` text,
   `Category` text,
   `AdditionalNotes` text,
   `Quantity` int DEFAULT NULL,
   `NVL` text,
   `BEN` text,
   `CurrentState` text,
   `LamPartNumber` text,
   `RecordIDNumber` int DEFAULT NULL,
   `SerialNumber` text,
   `materialPK` int NOT NULL AUTO_INCREMENT,
   `Location` varchar(255) DEFAULT NULL,
   PRIMARY KEY (`materialPK`)
 ) ENGINE=InnoDB AUTO_INCREMENT=12 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci
